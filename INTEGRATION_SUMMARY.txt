--------------------------------------------------------------------------------
MODULE API
--------------------------------------------------------------------------------

import("3dengine") returns:

FUNCTIONS:
  world_create() -> int
  world_destroy(world_id) -> nil
  world_run(world_id, update_fn, draw_fn) -> nil
  world_set_camera(world_id, pos, target, up) -> nil
  world_get_camera(world_id) -> {position, target, up}
  world_query(world_id, component_type) -> array
  mesh_make(world_id, primitive, params) -> int
  material_make(world_id, params) -> int
  texture_make(world_id, path_or_data) -> int
  mesh_make_from_obj(world_id, obj_path) -> int
  world_set_lighting(world_id, params) -> nil
  world_get_lighting(world_id) -> lighting_state
  world_parallel_runtime_stats(world_id) -> stats

CONSTANTS:
  VERSION = "0.1.0"
  CONVENTIONS = {handedness, world_up, ndc_depth, matrix_layout}
